        function puntitos(donde, caracter) {//Convertir numero en formato moneda a medida que se escribe
            pat = /[\*,\+,\(,\),\?,\,$,\[,\],\^]/;
            valor = donde.value;
            largo = valor.length;
            crtr = true;
            if (isNaN(caracter) || pat.test(caracter) == true) {
                if (pat.test(caracter) == true) {
                    caracter = '\'' + caracter;
                }
                carcter = new RegExp(caracter, "g");
                valor = valor.replace(carcter, "");
                donde.value = valor;
                crtr = false;
            }
            else {
                var nums = new Array();
                cont = 0;
                for (m = 0; m < largo; m++) {
                    if (valor.charAt(m) == "." || valor.charAt(m) == " ") {
                        continue;
                    }
                    else {
                        nums[cont] = valor.charAt(m);
                        cont++;
                    }
                }
            }
            var cad1 = "", cad2 = "", tres = 0;
            if (largo > 3 && crtr == true) {
                for (k = nums.length - 1; k >= 0; k--) {
                    cad1 = nums[k];
                    cad2 = cad1 + cad2;
                    tres++;
                    if ((tres % 3) == 0) {
                        if (k != 0) {
                            cad2 = "." + cad2;
                        }
                    }
                }
                donde.value = cad2;
            }
        }

            sideNav: function () {
                if ($('.pageslideBg').length == 0) {
                    $('<div />').attr('class', 'pageslideBg').appendTo($('body'));
                }
                var $btn = $(this),
                    $pageslide = $($btn.attr("href")),
                    $pageslideBg = $('.pageslideBg'),
                    _width = $pageslide.width();
                $btn.click(function () {
                    $pageslideBg.show();
                    $pageslide.css({'display': 'block'}).animate({'left': 0});
                    return false;
                });
                $pageslideBg.click(function () {
                    $(this).hide();
                    $pageslide.animate({'left': _width * -1 + 'px'}, function () {
                        $(this).hide();
                    });
                    return false;
                });

                return this;
            }

        });
    })(jQuery);

                        <p><strong>CRÉDITO DESEMBOLSADO</strong></p>
                        <p>Un crédito es desembolsado luego de haber cancelado todas las casas comerciales, haber
                            realizado el giro al docente y finalmente haya sido legalizado en el sistema su ingreso a
                            cartera.</p>

                        <p>Es la sumatoria del valor de los créditos que los vendedores radican  y que cumplen con

                        <p>Es el proceso mediante el cual se ingresa al sistema FINSONET las características del
                            crédito y la información del cliente disponible para entrar al primer filtro de
                            revisión y que en el sistema aparece con la etiqueta de "DOCUMENTOS".</p>
                        <br>
                        <p><strong>FECHA DE CIERRE</strong></p>
                        <p>La fecha de cierre será el último día hábil de cada mes.</p>
                        <br>
                        <p><strong>LIQUIDACIÓN PRELIMINAR</strong></p>
                        <p>Liquidación de comisiones que se hará el primer día hábil de cada mes
                            en la cual se calcularán y pagarán las comisiones de los créditos
                            desembolsados hasta la fecha de cierre. No se incluyen en este pago los créditos que se
                            encuentre aprobados.</p>
                        <br>
                        <p><strong>LIQUIDACIÓN DEFINITIVA</strong></p>
                        <p>Liquidación que se hará un día hábil después del cierre
                            contable, en la cual se calcularán las comisiones de los créditos que hayan sido
                            aprobados hasta la fecha de cierre de mes anterior y que a su vez hayan sido desembolsados hasta
                            el día de cierre contable. </p>
                        <br>
                        <p><strong>2. CONSIDERACIONES</strong></p>
                        <br>
                        <ul>
                            <li>A los asesores nuevos, durante los dos primeros meses de labor en la empresa, el % de
                                pago de comisión será del 1%. A partir del tercer mes se le pagará
                                según la tabla. </li>
                            <li>El proceso de liquidación para el pago de las comisiones se llevará a cabo en
                                dos partes: una liquidación preliminar y una liquidación definitiva.</li>
                            <li>Para liquidación preliminar solo se tendrán en cuenta la colocación y
                                el número de créditos que hayan sido desembolsados hasta la fecha de cierre.</li>
                            <li>Para liquidación definitiva solo se tendrán en cuenta los créditos
                                aprobados hasta la fecha de cierre de mes y que logren su desembolso hasta la fecha de
                                cierre contable.</li>
                            <li>Si un crédito es aprobado antes de la fecha de cierre de mes y no logra su desembolso
                                antes de la fecha de cierre contable, no se incluirá en la liquidación
                                definitiva para el pago de comisión de ese mes, quedando para la liquidación
                                de comisiones del mes siguiente.</li>
                            <li>Para el pago de la liquidación definitiva se totalizará nuevamente la
                                colocación y el número de créditos desembolsados por asesor según
                                la tabla de comisiones. Por lo tanto, solo se pagará en la liquidación
                                definitiva el excedente luego de haber descontado el pago realizado en la liquidación
                                preliminar.</li>
